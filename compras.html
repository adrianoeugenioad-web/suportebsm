async function salvarNaNuvem(){ const data = document.getElementById("dataReferencia").value; if(!data) return false; const payload = montarDadosTela(); const compradorUsuario = cotacaoAtual?.comprador_usuario || (usuarioEhAdmin() ? document.getElementById("filtroComprador").value : usuarioSessao.usuario); const compradorNome = cotacaoAtual?.comprador_nome || document.getElementById("compradorNome").value || usuarioSessao.nome; let numeroCotacao = cotacaoAtual?.numero_cotacao; if(!numeroCotacao){ numeroCotacao = await obterProximoNumeroCotacao(data, tipoCotacaoAtual); } const registro = { numero_cotacao: numeroCotacao, tipo_cotacao: tipoCotacaoAtual, data_referencia: data, comprador_usuario: compradorUsuario, comprador_nome: compradorNome, comprador_whatsapp: payload.compradorWhatsapp || "", responsavel: payload.responsavel || "", observacao: payload.observacao || "", status_aprovacao: cotacaoAtual?.status_aprovacao || "PENDENTE", aprovado_por: cotacaoAtual?.aprovado_por || null, aprovado_em: cotacaoAtual?.aprovado_em || null, dados: payload }; let result; if(cotacaoAtual?.id){ result = await client .from("cotacoes_bsm") .update(registro) .eq("id", cotacaoAtual.id) .select() .maybeSingle(); } else { result = await client .from("cotacoes_bsm") .insert(registro) .select() .maybeSingle(); } if(result.error){ alert("Erro ao salvar: " + result.error.message); return false; } cotacaoAtual = result.data; document.getElementById("numeroCotacao").innerText = cotacaoAtual.numero_cotacao; return true; }
